POST /api/shorten

Create a shortened URL

  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "Content-Type: application/json" \

// JavaScript/Node.js Example
const token = 'YOUR_TOKEN_HERE';

async function shortenUrl(originalUrl, customCode = null) {
  const response = await fetch('https://jmplnk.uk/api/shorten', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      url: originalUrl,
      shortCode: customCode
    })
  });
  
  const data = await response.json();
  console.log('Shortened URL:', data.url);
  return data;
}

// Usage
shortenUrl('https://example.com/long/url', 'mylink');
# Python Example
import requests

token = 'YOUR_TOKEN_HERE'
headers = {
    'Authorization': f'Bearer {token}',
    'Content-Type': 'application/json'
}

def shorten_url(original_url, custom_code=None):
    payload = {'url': original_url}
    if custom_code:
        payload['shortCode'] = custom_code
    
    response = requests.post(
        'https://jmplnk.uk/api/shorten',
        headers=headers,
        json=payload
    )
    
    data = response.json()
    print(f"Shortened URL: {data['url']}")
    return data

# Usage
shorten_url('https://example.com/long/url', 'mylink')

// PHP Example
$token = 'YOUR_TOKEN_HERE';

function shortenUrl($originalUrl, $customCode = null) {
    global $token;
    
    $data = ['url' => $originalUrl];
    if ($customCode) {
        $data['shortCode'] = $customCode;
    }
    
    $ch = curl_init('https://jmplnk.uk/api/shorten');
    curl_setopt($ch, CURLOPT_POST, 1);
    curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Authorization: Bearer ' . $token,
        'Content-Type: application/json'
    ]);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    
    $response = curl_exec($ch);
    curl_close($ch);
    
    $result = json_decode($response, true);
    echo "Shortened URL: " . $result['url'] . "\n";
    return $result;
}

// Usage
shortenUrl('https://example.com/long/url', 'mylink');
?>
